📝 유용한 SQL 쿼리

호텔 운영 현황

SELECT h.name as hotel_name, COUNT(r.id) as total_reservations, SUM(CASE WHEN r.status = 1 THEN 1 ELSE 0 END) as confirmed, SUM(r.total_amount) as total_revenue FROM hotels h LEFT JOIN reservations r ON h.id = r.hotel_id GROUP BY h.id, h.name;

월별 매출 통계

SELECT strftime('%Y-%m', created_at) as month, COUNT(*) as bookings, SUM(total_amount) as revenue, AVG(total_amount) as avg_booking_value FROM reservations WHERE status = 1 -- 1: 확정 GROUP BY strftime('%Y-%m', created_at) ORDER BY month DESC LIMIT 12;

인기 객실 타입

SELECT rt.name as room_type, rt.base_price, COUNT(r.id) as booking_count, AVG(r.total_amount) as avg_revenue FROM room_types rt LEFT JOIN reservations r ON rt.id = r.room_type_id WHERE r.status = 1 -- 1: 확정 GROUP BY rt.id, rt.name ORDER BY booking_count DESC;
